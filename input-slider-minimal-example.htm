Moving the slider changes the textbox; changing the text box moves the slider
Minimal Script Required... 

<input id="slider1" type="range" min="0" max="20" step="1" 
    onchange="setTextValue('slider1','rangeValue1')" />

<input id="rangeValue1" type="number" autofocus min="0" max="20" step="1" size="2" 
    onchange="setToolValue('slider1','rangeValue1')" />

<script>
    function setTextValue(sliderID, textbox) {
        var x = document.getElementById(textbox);
        var y = document.getElementById(sliderID);
        x.value = y.value;
    }

    function setToolValue(sliderID, textbox) {
        var x = document.getElementById(textbox);
        var y = document.getElementById(sliderID);
        y.value = x.value;
    }

    window.onload = function () {
        setTextValue('slider1', 'rangeValue1');
    }
</script>
